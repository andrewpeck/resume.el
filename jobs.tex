\newcommand{\project}[2]
{
  \setlength{\topsep}{3pt}%
  \setlength{\partopsep}{5pt}%
  \begin{tabbing}
    \large \textbf{#1} \` % \normalsize{\textit{#2}}
  \end{tabbing}
}

\newcommand{\job}[4]
{
  \setlength{\topsep}{3pt}%
  \setlength{\partopsep}{0pt}%
  \begin{tabbing}
    \textbf{#3} \` \small{#2}
    \ifthenelse {\equal{#1}{} \and \equal{#4}{}}
    {}
    {\\{\normalsize #1} \` \normalsize \small{#4}}
  \end{tabbing}
}
